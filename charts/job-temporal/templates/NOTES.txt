BEFORE the application will work you must create the required
Kubernetes secrets referenced in your values:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  # Application database
  kubectl create secret generic <database-secret> \
    --from-literal=DATABASE_URL="postgres://user:pass@host/db?sslmode=disable"

  # GitHub webhook secret (for the server component)
  kubectl create secret generic <server-secret> \
    --from-literal=GITHUB_WEBHOOK_SECRET="<your-webhook-secret>"

  # Temporal database credentials
  kubectl create secret generic <temporal-secret> \
    --from-literal=POSTGRES_USER="temporal" \
    --from-literal=POSTGRES_PWD="<password>" \
    --from-literal=POSTGRES_SEEDS="<postgres-hostname>"

  # Worker credentials
  kubectl create secret generic <worker-secret> \
    --from-literal=ANTHROPIC_API_KEY="<key>" \
    --from-literal=OPENAI_API_KEY="<key>" \
    --from-literal=GITHUB_APP_ID="<id>" \
    --from-literal=GITHUB_INSTALL_ID="<id>" \
    --from-literal=GITHUB_APP_PRIVATE_KEY="$(cat github-app.pem)" \
    --from-literal=AWS_ACCESS_KEY_ID="<key>" \
    --from-literal=AWS_SECRET_ACCESS_KEY="<key>" \
    --from-literal=AWS_REGION="auto" \
    --from-literal=AWS_ENDPOINT_URL="<r2-endpoint>" \
    --from-literal=R2_BUCKET="<bucket>" \
    --from-literal=R2_PUBLIC_BASE_URL="<url>"
